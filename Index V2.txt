<script>
    const topics = {
        'reading-writing': [
            'Craft and Structure',
            'Information and Ideas',
            'Standard English Conventions',
            'Expression of Ideas'
        ],
        'math': [
            'Algebra',
            'Advanced Math',
            'Problem-Solving and Data Analysis',
            'Geometry and Trigonometry'
        ]
    };

    let currentQuestion = null;

    function switchTab(tabName) {
        // Hide all tabs
        document.querySelectorAll('.tab-content').forEach(tab => {
            tab.classList.remove('active');
        });
        
        // Remove active class from all nav tabs
        document.querySelectorAll('.nav-tab').forEach(tab => {
            tab.classList.remove('active');
        });
        
        // Show selected tab
        document.getElementById(tabName).classList.add('active');
        
        // Add active class to clicked nav tab
        event.target.classList.add('active');
    }

    function updateTopics() {
        const section = document.getElementById('section-select').value;
        const topicSelect = document.getElementById('topic-select');
        const generateBtn = document.getElementById('generate-btn');
        
        topicSelect.innerHTML = '<option value="">-- Select a Topic --</option>';
        
        if (section && topics[section]) {
            topicSelect.disabled = false;
            topics[section].forEach(topic => {
                const option = document.createElement('option');
                option.value = topic.toLowerCase().replace(/\s+/g, '-');
                option.textContent = topic;
                topicSelect.appendChild(option);
            });
        } else {
            topicSelect.disabled = true;
            generateBtn.disabled = true;
        }
        
        // Clear previous question
        document.getElementById('question-display').innerHTML = '';
    }

    document.getElementById('topic-select').addEventListener('change', function() {
        const generateBtn = document.getElementById('generate-btn');
        generateBtn.disabled = !this.value;
    });

    async function generateQuestion() {
        const section = document.getElementById('section-select').value;
        const topic = document.getElementById('topic-select').value;
        const displayDiv = document.getElementById('question-display');
        
        if (!section || !topic) {
            alert('Please select both a section and topic');
            return;
        }
        
        displayDiv.innerHTML = '<div class="loading">Generating your practice question...</div>';
        
        try {
            const response = await fetch('https://api.anthropic.com/v1/messages', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({
                    model: 'claude-sonnet-4-20250514',
                    max_tokens: 1000,
                    messages: [{
                        role: 'user',
                        content: `Generate an SAT practice question for the ${section} section, topic: ${topic}. 

The question should be equivalent in difficulty and style to questions from the official SAT Question Bank, but NOT identical to any existing question.

Return ONLY valid JSON in this exact format:
{
  "question": "The question text here",
  "options": ["A) First option", "B) Second option", "C) Third option", "D) Fourth option"],
  "correct": 0,
  "explanation": "Detailed explanation of why the answer is correct"
}

The "correct" field should be the index (0-3) of the correct answer.`
                    }]
                })
            });
            
            const data = await response.json();
            
            if (data.content && data.content[0] && data.content[0].text) {
                const jsonText = data.content[0].text.replace(/```json\n?/g, '').replace(/```\n?/g, '').trim();
                currentQuestion = JSON.parse(jsonText);
                displayQuestion(currentQuestion);
            } else {
                throw new Error('Invalid response format');
            }
        } catch (error) {
            console.error('Error:', error);
            displayDiv.innerHTML = '<div class="loading" style="color: red;">Error generating question. Please try again.</div>';
        }
    }

    function displayQuestion(question) {
        const displayDiv = document.getElementById('question-display');
        
        let html = '<div class="question-container">';
        html += `<div class="question-text">${question.question}</div>`;
        
        question.options.forEach((option, index) => {
            html += `<div class="answer-option" onclick="selectAnswer(${index})" data-index="${index}">
                ${option}
            </div>`;
        });
        
        html += '<button class="check-answer-btn" onclick="checkAnswer()">Check Answer</button>';
        html += '</div>';
        
        displayDiv.innerHTML = html;
    }

    function selectAnswer(index) {
        document.querySelectorAll('.answer-option').forEach(opt => {
            opt.classList.remove('selected');
        });
        
        document.querySelector(`[data-index="${index}"]`).classList.add('selected');
    }

    function checkAnswer() {
        const selected = document.querySelector('.answer-option.selected');
        
        if (!selected) {
            alert('Please select an answer first');
            return;
        }
        
        const selectedIndex = parseInt(selected.getAttribute('data-index'));
        const options = document.querySelectorAll('.answer-option');
        
        options.forEach((opt, index) => {
            opt.style.pointerEvents = 'none';
            if (index === currentQuestion.correct) {
                opt.classList.add('correct');
            } else if (index === selectedIndex && index !== currentQuestion.correct) {
                opt.classList.add('incorrect');
            }
        });
        
        const displayDiv = document.getElementById('question-display');
        const explanation = document.createElement('div');
        explanation.className = 'explanation';
        explanation.innerHTML = `<strong>Explanation:</strong><br>${currentQuestion.explanation}`;
        displayDiv.querySelector('.question-container').appendChild(explanation);
        
        document.querySelector('.check-answer-btn').style.display = 'none';
    }
</script>
